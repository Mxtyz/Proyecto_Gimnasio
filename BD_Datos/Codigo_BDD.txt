CREATE TABLE Usuario(
Rut varchar(15) PRIMARY KEY,
Nombre varchar(30),
Apellido varchar(30),
Correo varchar(50),
Contrase√±a varchar(15),
TipoUsuario varchar(20),
Telefono varchar(10)
);

CREATE TABLE Sala(
IdSala integer PRIMARY KEY,
Nombre varchar(30),
Capacidad smallint,
TipoSala varchar(30),
Disponible boolean
);

CREATE TABLE Clase(
IdClase integer PRIMARY KEY,
NombreClase varchar(20),
Descripcion varchar(255),
Duracion integer,
Cupos smallint
);

CREATE TABLE Horario(
IdHorario integer PRIMARY KEY,
Fecha date,
HoraInicio time,
HoraFin time,
Estado boolean,
RutUsuario varchar(15),
IdSala integer,
IdClase integer,
FOREIGN KEY (RutUsuario) REFERENCES Usuario(Rut),
FOREIGN KEY (IdSala) REFERENCES Sala(IdSala),
FOREIGN KEY (IdClase) REFERENCES Clase(IdClase)
);

CREATE TABLE Notificacion(
IdNotificacion integer PRIMARY KEY,
Mensaje varchar(255),
TipoNotificacion varchar(30),
FechaEnvio date,
Estado Boolean,
RutUsuario varchar(15),
FOREIGN KEY (RutUsuario) REFERENCES Usuario(Rut)
);

CREATE TABLE Reserva(
IdReserva integer PRIMARY KEY,
FechaReserva date,
EstadoReserva boolean,
RutUsuario varchar(15),
IdHorario integer,
FOREIGN KEY (RutUsuario) REFERENCES Usuario(Rut),
FOREIGN KEY (IdHorario) REFERENCES Horario(IdHorario)
);
